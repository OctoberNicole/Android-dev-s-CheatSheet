class Permutation(intArr: IntArray) {

    private var originItems: ArrayList<Int>
    private var numberOfPeoples = 0
    private var combination: IntArray
    private var combinationCount = 0
    private var executeWorkCount = 0

    init {
        numberOfPeoples = intArr.size
        combination = IntArray(numberOfPeoples)
        originItems = ArrayList(intArr.toList())
    }

    fun executeWork(temporaryItems: ArrayList<Int> = originItems, depth: Int = 0): Int {
        executeWorkCount++

        when (depth == numberOfPeoples) {
            true -> {
                combinationCount++
                println("${combinationCount}번째 조합")
                println(combination.joinToString(", ") + System.lineSeparator())
            }
            false -> {
                for (i in 0 until numberOfPeoples-depth) {
                    combination[depth] = temporaryItems.removeAt(i)
                    executeWork(temporaryItems, depth.plus(1))
                    temporaryItems.add(i, combination[depth])
                }
            }
        }

        return combinationCount
    }
}

fun main(args: Array<String>) {
    println(Permutation(intArrayOf(20, 8, 10, 1, 4, 15)).executeWork())
}
